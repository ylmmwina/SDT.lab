114class Dijkstra {
115public:

126 void run(const Graph<TNode, WeightedEdge>& g, const TNode& start) {
127 // Ініціалізація
128 dist.clear(); parent.clear();
129 for (auto& [u, _] : g.data()) dist[u] = std::numeric_limits<double>::infinity();
130 if (!g.hasNode(start)) return;
131 dist[start] = 0.0;
132
133 // мін-купа (черга з парою (distance, node))
134 using QItem = std::pair<double, TNode>;
135 std::priority_queue<QItem, std::vector<QItem>, std::greater<QItem>> pq;
136 pq.push({0.0, start});
137
138 while (!pq.empty()) {
139 auto [du, u] = pq.top(); pq.pop();
140 if (du != dist[u]) continue; // пропускаємо застарілі значення
141
142 // перебираємо сусідів
143 auto it = g.data().find(u);
144 if (it == g.data().end()) continue;
145 for (auto& [v, w] : it->second) {
146 double nd = du + w.weight;
147 if (nd < dist[v]) { // релаксація
148 dist[v] = nd;
149 parent[v] = u;
150 pq.push({nd, v});
151 }
152 }
153 }
154 }
155
162 std::vector<TNode> getPathTo(const TNode& start, const TNode& target) const {
163 std::vector<TNode> path;
164 if (!dist.count(target) || dist.at(target) == std::numeric_limits<double>::infinity())
165 return path; // пусто: шляху немає
166 TNode cur = target;
167 while (cur != start) {
168 path.push_back(cur);
169 auto it = parent.find(cur);
170 if (it == parent.end()) { path.clear(); return path; }
171 cur = it->second;
172 }
173 path.push_back(start);
174 std::reverse(path.begin(), path.end());
175 return path;
176 }
